173/**
174 * Attempts to handle a keyboard event; returns true if the event was handled, false otherwise. Considered protected;
175 * should only be used within this package and by subclasses.
176 * @param {goog.events.KeyEvent} e Key event to handle.
177 * @return {boolean} Whether the key event was handled.
178 * @protected
179 * @override
180 */
181zz.ui.mdl.Checkbox.prototype.handleKeyEventInternal = function( e ){
182
183 if( e.keyCode === goog.events.KeyCodes.ENTER ){
184
185 this.setInputValue( !this.getInputValue( ) );
186 this.changeListener_( );
187 return true;
188 }
189 return false;
190};
