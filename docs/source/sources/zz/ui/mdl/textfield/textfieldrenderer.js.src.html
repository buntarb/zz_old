67/**
68 * @param {zz.ui.mdl.TextField} control
69 * @param {Element} element
70 * @override
71 */
72zz.ui.mdl.TextFieldRenderer.prototype.decorate = function( control, element ){
73
74 // Label element
75 control.setLabelElement( control.getDomHelper( ).getElementsByTagNameAndClass(
76
77 goog.dom.TagName.LABEL,
78 zz.ui.mdl.TextField.CSS.LABEL
79 ) );
80
81 // Input element
82 control.setInputElement( control.getDomHelper( ).getElementByClass(
83
84 zz.ui.mdl.TextField.CSS.INPUT
85 ) );
86
87 // Text aria element attributes
88 if( control.getInputElement( ).hasAttribute( zz.ui.mdl.TextField.CONST.MAX_ROWS_ATTRIBUTE ) ){
89
90 control.setMaxRows(
91
92 parseInt(
93
94 control.getInputElement( ).getAttribute( zz.ui.mdl.TextField.CONST.MAX_ROWS_ATTRIBUTE ),
95 10 ) );
96
97 if( isNaN( control.getMaxRows( ) ) ){
98
99 control.setMaxRows( zz.ui.mdl.TextField.CONST.NO_MAX_ROWS );
100 }
101 }
102 goog.dom.classlist.add( element, zz.ui.mdl.TextField.CSS.IS_UPGRADED );
103 return goog.base( this, 'decorate', control, element );
104};

122/**
123 * @param {zz.ui.mdl.TextField} control
124 */
125zz.ui.mdl.TextFieldRenderer.prototype.updateClasses = function( control ){
126
127 //noinspection JSUnresolvedFunction
128 if( control.isEnabled( ) ){
129
130 goog.dom.classlist.remove( control.getElement( ), zz.ui.mdl.TextField.CSS.IS_DISABLED );
131
132 }else{
133
134 goog.dom.classlist.add( control.getElement( ), zz.ui.mdl.TextField.CSS.IS_DISABLED );
135 }
136 if( control.getInputElement( ).validity ){
137
138 if( control.getInputElement( ).validity.valid ){
139
140 goog.dom.classlist.remove( control.getElement( ), zz.ui.mdl.TextField.CSS.IS_INVALID );
141
142 }else{
143
144 goog.dom.classlist.add( control.getElement( ), zz.ui.mdl.TextField.CSS.IS_INVALID );
145 }
146 }
147 if( control.getInputValue( ).length > 0 ){
148
149 goog.dom.classlist.add( control.getElement( ), zz.ui.mdl.TextField.CSS.IS_DIRTY );
150
151 }else{
152
153 goog.dom.classlist.remove( control.getElement( ), zz.ui.mdl.TextField.CSS.IS_DIRTY );
154 }
155};
