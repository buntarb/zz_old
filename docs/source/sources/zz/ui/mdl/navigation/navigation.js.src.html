181/**
182 * @override
183 */
184zz.ui.mdl.Navigation.prototype.decorateInternal = function( element ){
185
186 var children = element.childNodes;
187 var container = this.getDomHelper( ).createDom( goog.dom.TagName.DIV, {
188
189 'class': zz.ui.mdl.Navigation.CSS.CONTAINER
190 } );
191 goog.dom.insertSiblingBefore( container, element );
192 goog.dom.appendChild( container, element );
193 goog.base( this, 'decorateInternal', container );
194
195 // Find elements
196 if( !this.headerElement_ ||
197 !this.drawerElement_ ||
198 !this.bodyElement_ ) {
199
200 goog.array.forEach( children, function( child ){
201
202 if( goog.dom.classlist.contains( child, zz.ui.mdl.Navigation.CSS.HEADER ) ){
203
204 this.headerElement_ = child;
205 }
206 if( goog.dom.classlist.contains( child, zz.ui.mdl.Navigation.CSS.DRAWER ) ){
207
208 this.drawerElement_ = child;
209 }
210 if( goog.dom.classlist.contains( child, zz.ui.mdl.Navigation.CSS.CONTENT ) ){
211
212 this.bodyElement_ = child;
213 }
214 }, this );
215 }
216
217 // Patch header
218 if( this.headerElement_ ){
219
220 this.tabBarElement_ = goog.dom.getElementByClass(
221
222 zz.ui.mdl.Navigation.CSS.TAB_BAR,
223 this.headerElement_ );
224
225 if( goog.dom.classlist.contains(
226 this.headerElement_,
227 zz.ui.mdl.Navigation.CSS.HEADER_SEAMED ) ){
228
229 this.setMode( zz.ui.mdl.Navigation.MODE.SEAMED );
230
231 }else if( goog.dom.classlist.contains(
232 this.headerElement_,
233 zz.ui.mdl.Navigation.CSS.HEADER_WATERFALL ) ){
234
235 this.setMode( zz.ui.mdl.Navigation.MODE.WATERFALL );
236
237 }else if( goog.dom.classlist.contains(
238 this.headerElement_,
239 zz.ui.mdl.Navigation.CSS.HEADER_SCROLL ) ){
240
241 this.setMode( zz.ui.mdl.Navigation.MODE.SCROLL );
242 goog.dom.classlist.add( container, zz.ui.mdl.Navigation.CSS.HAS_SCROLLING_HEADER );
243 }
244 if( this.getMode( ) === zz.ui.mdl.Navigation.MODE.STANDARD ){
245
246 goog.dom.classlist.add( this.headerElement_, zz.ui.mdl.Navigation.CSS.CASTING_SHADOW );
247 if( this.tabBarElement_ ){
248
249 goog.dom.classlist.add( this.tabBarElement_, zz.ui.mdl.Navigation.CSS.CASTING_SHADOW );
250 }
251 }else if( this.getMode( ) === zz.ui.mdl.Navigation.MODE.SEAMED ||
252 this.getMode( ) === zz.ui.mdl.Navigation.MODE.SCROLL ){
253
254 goog.dom.classlist.remove( this.headerElement_, zz.ui.mdl.Navigation.CSS.CASTING_SHADOW );
255 if( this.tabBarElement_ ){
256
257 goog.dom.classlist.remove( this.tabBarElement_, zz.ui.mdl.Navigation.CSS.CASTING_SHADOW );
258 }
259 }else if( this.getMode( ) === zz.ui.mdl.Navigation.MODE.WATERFALL ){
260
261 this.setStyleOnScroll( );
262 }
263 }
264
265 // Patch drawer
266 if( this.drawerElement_ ){
267
268 this.drawerButtonElement_ = goog.dom.getElementByClass( zz.ui.mdl.Navigation.CSS.DRAWER_BTN, element );
269 if( !this.drawerButtonElement_ ) {
270
271 this.createDrawerButtonDom_( );
272 }
273 if( goog.dom.classlist.contains( this.drawerElement_, zz.ui.mdl.Navigation.CSS.ON_LARGE_SCREEN ) ){
274
275 //If drawer has ON_LARGE_SCREEN class then add it to the drawer toggle button as well.
276 goog.dom.classlist.add( this.drawerButtonElement_, zz.ui.mdl.Navigation.CSS.ON_LARGE_SCREEN );
277
278 }else if( goog.dom.classlist.contains( this.drawerElement_, zz.ui.mdl.Navigation.CSS.ON_SMALL_SCREEN ) ){
279
280 //If drawer has ON_SMALL_SCREEN class then add it to the drawer toggle button as well.
281 goog.dom.classlist.add( this.drawerButtonElement_, zz.ui.mdl.Navigation.CSS.ON_SMALL_SCREEN );
282 }
283 goog.dom.classlist.add( element, zz.ui.mdl.Navigation.CSS.HAS_DRAWER );
284
285 if( goog.dom.classlist.contains( element, zz.ui.mdl.Navigation.CSS.FIXED_HEADER ) ){
286
287 goog.dom.insertChildAt( this.headerElement_, this.drawerButtonElement_, 0 );
288
289 }else{
290
291 goog.dom.insertChildAt( element, this.drawerButtonElement_, 0 );
292 }
293 this.createObfuscatorDom_( );
294 goog.dom.appendChild( element, this.obfuscatorElement_ );
295 this.setStyleOnResize( element );
296 }
297
298 // Patch tabs
299 if( this.headerElement_ && this.tabBarElement_ ){
300
301 goog.dom.classlist.add( element, zz.ui.mdl.Navigation.CSS.HAS_TABS );
302 this.createTabBarDom_( );
303 }
304
305 // Final changes
306 goog.dom.classlist.add( element, zz.ui.mdl.Navigation.CSS.IS_UPGRADED );
307};

309/**
310 * Called when the component's element is known to be in the document. Anything using document.getElementById etc.
311 * should be done at this stage. If the component contains child components, this call is propagated to its children.
312 * @override
313 */
314zz.ui.mdl.Navigation.prototype.enterDocument = function( ){
315
316 goog.base( this, 'enterDocument' );
317
318 if( this.mode_ === zz.ui.mdl.Navigation.MODE.WATERFALL ){
319
320 this.getHandler( ).listenWithScope(
321
322 this.headerElement_,
323 goog.events.EventType.MOUSEUP,
324 this.headerClickListener_,
325 false,
326 this
327 );
328 this.getHandler( ).listenWithScope(
329
330 this.headerElement_,
331 goog.events.EventType.TRANSITIONEND,
332 this.headerTransitionEndListener_,
333 false,
334 this
335 );
336 // Add and remove shadows depending on scroll position.
337 // Also add/remove auxiliary class for styling of the compact version of the header.
338 this.getHandler( ).listenWithScope(
339
340 this.bodyElement_,
341 goog.events.EventType.SCROLL,
342 this.bodyScrollListener_,
343 false,
344 this
345 );
346 }
347 this.getHandler( ).listenWithScope(
348
349 this.drawerButtonElement_,
350 goog.events.EventType.MOUSEUP,
351 this.drawerToggleListener_,
352 false,
353 this
354 );
355 this.getHandler( ).listenWithScope(
356
357 this.obfuscatorElement_,
358 goog.events.EventType.MOUSEUP,
359 this.drawerToggleListener_,
360 false,
361 this
362 );
363 this.getHandler( ).listenWithScope(
364
365 this.getDomHelper( ).getWindow( ),
366 goog.events.EventType.RESIZE,
367 this.windowResizeListener_,
368 false,
369 this
370 );
371 if( this.headerElement_ && this.tabBarElement_ ){
372
373 this.getHandler( ).listenWithScope(
374
375 this.tabBarLeftButtonElement_,
376 goog.events.EventType.MOUSEUP,
377 this.tabBarLeftButtonClickListener_,
378 false,
379 this
380 );
381 this.getHandler( ).listenWithScope(
382
383 this.tabBarRightButtonElement_,
384 goog.events.EventType.MOUSEUP,
385 this.tabBarRightButtonClickListener_,
386 false,
387 this
388 );
389 this.getHandler( ).listenWithScope(
390
391 this.tabBarElement_,
392 goog.events.EventType.SCROLL,
393 this.tabBarScrollListener_,
394 false,
395 this
396 );
397 goog.array.forEach( this.tabsElements_, function( tab ){
398
399 this.getHandler( ).listenWithScope(
400
401 tab,
402 goog.events.EventType.MOUSEUP,
403 /** @this {zz.ui.mdl.Navigation} */
404 function( evt ){
405
406 evt.preventDefault( );
407
408 var href = tab.href.split( '#' )[ 1 ];
409 var panel = goog.dom.getElement( href );
410
411 //noinspection JSPotentiallyInvalidUsageOfThis
412 this.resetTabState( this.tabsElements_ );
413
414 //noinspection JSPotentiallyInvalidUsageOfThis
415 this.resetPanelState( this.panelsElements_ );
416
417 //noinspection JSPotentiallyInvalidUsageOfThis
418 this.setTabActive( tab );
419
420 //noinspection JSPotentiallyInvalidUsageOfThis
421 this.setPanelActive( panel );
422 },
423 false,
424 this
425 );
426 }, this );
427 }
428 this.ripples = [ ];
429 goog.array.forEach(
430
431 goog.dom.getElementsByClass( zz.ui.mdl.Navigation.CSS.RIPPLE_CONTAINER ),
432 function( ripple ){
433
434 this.ripples.push( new zz.ui.mdl.Ripple( ) );
435 this.addChild( this.ripples[ this.ripples.length - 1 ], false );
436 this.ripples[ this.ripples.length - 1 ].decorate( ripple );
437
438 }, this );
439};

442 * Deletes or nulls out any references to COM objects, DOM nodes, or other disposable objects. Classes that extend
443 * {@code goog.Disposable} should override this method. Not reentrant. To avoid calling it twice, it must only be
444 * called from the subclass' {@code disposeInternal} method. Everywhere else the public {@code dispose} method must
445 * be used.
446 * @inheritDoc
447 **/
448zz.ui.mdl.Navigation.prototype.disposeInternal = function( ){

461 this.obfuscatorElement_ = null;
462 goog.array.forEach( this.ripples, function( ripple ){
463
464 ripple.dispose( );
465 } );
466 this.ripples = [ ];
467};

709/**
710 * Update styles on window resize event.
711 * @param {Element} element
712 */
713zz.ui.mdl.Navigation.prototype.setStyleOnResize = function( element ){
714
715 //noinspection JSUnresolvedVariable
716 if( this.getDomHelper( ).getWindow( ).innerWidth <= zz.ui.mdl.Navigation.CONST.MAX_WIDTH ){
