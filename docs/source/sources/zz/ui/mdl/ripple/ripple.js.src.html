167

230/**
231 * Handle mouse/finger down on element.
232 * @param {goog.events.BrowserEvent} event
233 * @private
234 */
235zz.ui.mdl.Ripple.prototype.downHandler_ = function( event ){
236
237 if( !goog.style.getStyle( this.rippleElement_ ).width &&
238 !goog.style.getStyle( this.rippleElement_ ).height ){
239
240 var rect = goog.style.getBounds( this.element_ );
241 this.boundHeight = rect.height;
242 this.boundWidth = rect.width;
243 this.rippleSize_ = Math.sqrt( rect.width * rect.width + rect.height * rect.height ) * 2 + 2;
244 goog.style.setStyle( this.rippleElement_, {
245
246 'width': this.rippleSize_ + 'px',
247 'height': this.rippleSize_ + 'px'
248 } );
249 }
250 //noinspection JSValidateTypes
251 goog.dom.classlist.add( this.rippleElement_, zz.ui.mdl.Ripple.CSS.IS_VISIBLE );
252 if( event.type === goog.events.EventType.MOUSEDOWN && this.ignoringMouseDown_ ){
253
254 this.ignoringMouseDown_ = false;
255
256 }else{
257
258 if( event.type === goog.events.EventType.TOUCHSTART ){
259
260 this.ignoringMouseDown_ = true;
261 }
262 var frameCount = this.getFrameCount( );
263 if( frameCount > 0 ){
264
265 return;
266 }
267 this.setFrameCount( 1 );
268 //var bound = event.currentTarget.getBoundingClientRect( );
269 var bound = goog.style.getBounds( event.currentTarget );
270 var x;
271 var y;
272 // Check if we are handling a keyboard click.
273 if( event.clientX === 0 && event.clientY === 0 ){
274
275 x = Math.round( bound.width / 2 );
276 y = Math.round( bound.height / 2 );
277
278 }else{
279
280 // TODO (buntarb): Test this code.
281 var clientX = event.clientX ? event.clientX : event.getBrowserEvent( ).touches[ 0 ].clientX;
282 var clientY = event.clientY ? event.clientY : event.getBrowserEvent( ).touches[ 0 ].clientY;
283 x = Math.round( clientX - bound.left );
284 y = Math.round( clientY - bound.top );
285 }
286 this.setRippleXY( x, y );
287 this.setRippleStyles( true );
288 window.requestAnimationFrame( goog.bind( this.animationFrameHandler, this ) );
289 }
290};

292/**
293 * Handle mouse / finger up on element.
294 * @param {goog.events.BrowserEvent} event
295 * @private
296 */
297zz.ui.mdl.Ripple.prototype.upHandler_ = function( event ){
298
299 // Don't fire for the artificial "mouseup" generated by a double-click.
300 if( event && event.getBrowserEvent( ).detail !== 2 ){
301
302 //noinspection JSValidateTypes
303 goog.dom.classlist.remove( this.rippleElement_, zz.ui.mdl.Ripple.CSS.IS_VISIBLE );
304 }
305 // Allow a repaint to occur before removing this class, so the animation
306 // shows for tap events, which seem to trigger a mouseup too soon after
307 // mousedown.
308 goog.Timer.callOnce( function( ){
309
310 //noinspection JSPotentiallyInvalidUsageOfThis,JSValidateTypes
311 goog.dom.classlist.remove( this.rippleElement_, zz.ui.mdl.Ripple.CSS.IS_VISIBLE );
312
313 }, 0, this );
314};

340/**
341 * Sets the ripple styles.
342 * @param {boolean} start whether or not this is the start frame.
343 */
344zz.ui.mdl.Ripple.prototype.setRippleStyles = function( start ){
345
346 if( this.rippleElement_ !== null ){
347
348 var transformString;
349 var scale;
350 var offset = 'translate(' + this.x_ + 'px, ' + this.y_ + 'px)';
351 if( start ){
352
353 scale = zz.ui.mdl.Ripple.CONST.INITIAL_SCALE;
354
355 }else{
356
357 scale = zz.ui.mdl.Ripple.CONST.FINAL_SCALE;
358 if( this.centeringFlag_ ){
359
360 offset = 'translate(' + this.boundWidth / 2 + 'px, ' + this.boundHeight / 2 + 'px)';
361 }
362 }
363 transformString = 'translate(-50%, -50%) ' + offset + scale;
364 goog.style.setStyle( this.rippleElement_, {
365
366 'webkitTransform': transformString,
367 'msTransform': transformString,
368 'transform': transformString
369 } );
370 if( start ){
371
372 goog.dom.classlist.remove( /** @type {Element} */( this.rippleElement_ ), zz.ui.mdl.Ripple.CSS.IS_ANIMATING );
373
374 }else{
375
376 this.rippleElement_.classList.add(zz.ui.mdl.Ripple.CSS.IS_ANIMATING);
377 }
378 }
