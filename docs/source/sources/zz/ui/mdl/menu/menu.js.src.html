138/**
139 * Called when the component's element is known to be in the document. Anything using document.getElementById etc.
140 * should be done at this stage. If the component contains child components, this call is propagated to its children.
141 * @override
142 */
143zz.ui.mdl.Menu.prototype.enterDocument = function( ){
144
145 goog.base( this, 'enterDocument' );
146
147 this.setVisible( false );
148 this.getHandler( ).listenWithScope(
149
150 this.listElement_,
151 goog.events.EventType.TRANSITIONEND,
152 this.animationEndListener_,
153 false,
154 this
155 );
156 goog.array.forEach( this.getItemsElements( ), function( item ){
157
158 if( goog.dom.classlist.contains( item, zz.ui.mdl.MenuItem.CSS.RIPPLE_EFFECT ) ){
159
160 var i = this.ripples_.push( new zz.ui.mdl.Ripple( ) );
161 this.ripples_[ i - 1 ].decorate( item );
162 }
163 }, this );
164
165};

167/**
168 * Deletes or nulls out any references to COM objects, DOM nodes, or other disposable objects. Classes that extend
169 * {@code goog.Disposable} should override this method. Not reentrant. To avoid calling it twice, it must only be
170 * called from the subclass' {@code disposeInternal} method. Everywhere else the public {@code dispose} method must
171 * be used.
172 * @inheritDoc
173 **/
174zz.ui.mdl.Menu.prototype.disposeInternal = function( ){
175
176 goog.base( this, 'disposeInternal' );
177 this.getHandler( ).dispose( );
178 zz.service.Popup.getInstance( ).removeClosable( this );
179 goog.array.forEach( this.ripples_, function( ripple ){
180
181 ripple.dispose( );
182 } );
183};
