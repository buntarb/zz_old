18/**********************************************************************************************************************

21goog.provide( 'zz.ui.mdl.DemoAnimation' );
22
23/**********************************************************************************************************************
24 * Dependencies section *
25 **********************************************************************************************************************/
26goog.require( 'zz.ui.mdl.componentHandler' );
27
28/**
29 * Class constructor for Animation MDL component.
30 * Implements MDL component design pattern defined at:
31 * https://github.com/jasonmayes/mdl-component-design-pattern
32 * @param {HTMLElement} element The element that will be upgraded.
33 */
34function DemoAnimation(element) {
35 'use strict';
36
37 this.element_ = element;
38 this.position_ = this.Constant_.STARTING_POSITION;
39 this.movable_ = this.element_.querySelector('.' + this.CssClasses_.MOVABLE);
40 // Initialize instance.
41 this.init();
42}
43
44/**
45 * Store strings for class names defined by this component that are used in
46 * JavaScript. This allows us to simply change it in one place should we
47 * decide to modify at a later date.
48 * @enum {string}
49 * @private
50 */
51DemoAnimation.prototype.CssClasses_ = {
52 MOVABLE: goog.getCssName( 'demo-animation__movable' ),
53 POSITION_PREFIX: goog.getCssName( 'demo-animation--position-' ),
54 FAST_OUT_SLOW_IN: goog.getCssName( 'mdl-animation--fast-out-slow-in' ),
55 LINEAR_OUT_SLOW_IN: goog.getCssName( 'mdl-animation--linear-out-slow-in' ),
56 FAST_OUT_LINEAR_IN: goog.getCssName( 'mdl-animation--fast-out-linear-in' )
57};
58
59/**
60 * Store constants in one place so they can be updated easily.
61 * @enum {string | number}
62 * @private
63 */
64DemoAnimation.prototype.Constant_ = {
65 STARTING_POSITION: 0,
66 // Which animation to use for which state. Check demo.css for an explanation.
67 ANIMATIONS: [
68 DemoAnimation.prototype.CssClasses_.FAST_OUT_LINEAR_IN,
69 DemoAnimation.prototype.CssClasses_.LINEAR_OUT_SLOW_IN,
70 DemoAnimation.prototype.CssClasses_.FAST_OUT_SLOW_IN,
71 DemoAnimation.prototype.CssClasses_.FAST_OUT_LINEAR_IN,
72 DemoAnimation.prototype.CssClasses_.LINEAR_OUT_SLOW_IN,
73 DemoAnimation.prototype.CssClasses_.FAST_OUT_SLOW_IN
74 ]
75};
76
77/**
78 * Handle click of element.
79 * @param {Event} event The event that fired.
80 * @private
81 */
82DemoAnimation.prototype.handleClick_ = function(event) {
83 'use strict';
84
85 this.movable_.classList.remove(this.CssClasses_.POSITION_PREFIX + this.position_);
86 this.movable_.classList.remove(this.Constant_.ANIMATIONS[this.position_]);
87 this.position_++;
88 if (this.position_ > 5) {
89 this.position_ = 0;
90 }
91 this.movable_.classList.add(this.Constant_.ANIMATIONS[this.position_]);
92 this.movable_.classList.add(this.CssClasses_.POSITION_PREFIX + this.position_);
93};
94
95/**
96 * Initialize element.
97 */
98DemoAnimation.prototype.init = function() {
99 'use strict';
100
101 if (this.element_) {
102 if (!this.movable_) {
103 console.error('Was expecting to find an element with class name ' +
104 this.CssClasses_.MOVABLE + ' inside of: ', this.element_);
105 return;
106 }
107
108 this.element_.addEventListener('click', this.handleClick_.bind(this));
109 }
110};
111
112// The component registers itself. It can assume componentHandler is available
113// in the global scope.
114componentHandler.register({
115 constructor: DemoAnimation,
116 classAsString: 'DemoAnimation',
117 cssClass: goog.getCssName( 'demo-js-animation' )
118});
