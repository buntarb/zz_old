21goog.provide( 'zz.ui.mdl.MaterialDataTable' );

23/**********************************************************************************************************************
24 * Dependencies section *
25 **********************************************************************************************************************/
26goog.require( 'zz.ui.mdl.componentHandler' );
27
28(function() {
29 'use strict';
30
31 /**
32 * Class constructor for Data Table Card MDL component.
33 * Implements MDL component design pattern defined at:
34 * https://github.com/jasonmayes/mdl-component-design-pattern
35 *
36 * @constructor
37 * @param {HTMLElement} element The element that will be upgraded.
38 */
39 var MaterialDataTable = function MaterialDataTable(element) {
40 this.element_ = element;
41
42 // Initialize instance.
43 this.init();
44 };
45
46 window['MaterialDataTable'] = MaterialDataTable;
47
48 /**
49 * Store constants in one place so they can be updated easily.
50 *
51 * @enum {string | number}
52 * @private
53 */
54 MaterialDataTable.prototype.Constant_ = {
55 // None at the moment.
56 };
57
58 /**
59 * Store strings for class names defined by this component that are used in
60 * JavaScript. This allows us to simply change it in one place should we
61 * decide to modify at a later date.
62 *
63 * @enum {string}
64 * @private
65 */
66 MaterialDataTable.prototype.CssClasses_ = {
67 DATA_TABLE: goog.getCssName( 'mdl-data-table' ),
68 SELECTABLE: goog.getCssName( 'mdl-data-table--selectable' ),
69 SELECT_ELEMENT: goog.getCssName( 'mdl-data-table__select' ),
70 IS_SELECTED: goog.getCssName( 'is-selected' ),
71 IS_UPGRADED: goog.getCssName( 'is-upgraded' )
72 };
73
74 /**
75 * Generates and returns a function that toggles the selection state of a
76 * single row (or multiple rows).
77 *
78 * @param {Element} checkbox Checkbox that toggles the selection state.
79 * @param {HTMLElement} row Row to toggle when checkbox changes.
80 * @param {(Array<Object>|NodeList)=} opt_rows Rows to toggle when checkbox changes.
81 * @private
82 */
83 MaterialDataTable.prototype.selectRow_ = function(checkbox, row, opt_rows) {
84 if (row) {
85 return function() {
86 if (checkbox.checked) {
87 row.classList.add(this.CssClasses_.IS_SELECTED);
88 } else {
89 row.classList.remove(this.CssClasses_.IS_SELECTED);
90 }
91 }.bind(this);
92 }
93
94 if (opt_rows) {
95 return function() {
96 var i;
97 var el;
98 if (checkbox.checked) {
99 for (i = 0; i < opt_rows.length; i++) {
100 el = opt_rows[i].querySelector('td').querySelector( '.' + goog.getCssName( 'mdl-checkbox' ) );
101 el['MaterialCheckbox'].check();
102 opt_rows[i].classList.add(this.CssClasses_.IS_SELECTED);
103 }
104 } else {
105 for (i = 0; i < opt_rows.length; i++) {
106 el = opt_rows[i].querySelector('td').querySelector( '.' + goog.getCssName( 'mdl-checkbox' ));
107 el['MaterialCheckbox'].uncheck();
108 opt_rows[i].classList.remove(this.CssClasses_.IS_SELECTED);
109 }
110 }
111 }.bind(this);
112 }
113 };
114
115 /**
116 * Creates a checkbox for a single or or multiple rows and hooks up the
117 * event handling.
118 *
119 * @param {HTMLElement} row Row to toggle when checkbox changes.
120 * @param {(Array<Object>|NodeList)=} opt_rows Rows to toggle when checkbox changes.
121 * @private
122 */
123 MaterialDataTable.prototype.createCheckbox_ = function(row, opt_rows) {
124 var label = document.createElement('label');
125 var labelClasses = [
126 goog.getCssName( 'mdl-checkbox' ),
127 goog.getCssName( 'mdl-js-checkbox' ),
128 goog.getCssName( 'mdl-js-ripple-effect' ),
129 this.CssClasses_.SELECT_ELEMENT
130 ];
131 label.className = labelClasses.join(' ');
132 var checkbox = document.createElement('input');
133 checkbox.type = 'checkbox';
134 checkbox.classList.add( goog.getCssName( 'mdl-checkbox__input' ) );
135 checkbox.addEventListener('change', this.selectRow_(checkbox, row, opt_rows));
136 label.appendChild(checkbox);
137 componentHandler.upgradeElement(label, 'MaterialCheckbox');
138 return label;
139 };
140
141 /**
142 * Initialize element.
143 */
144 MaterialDataTable.prototype.init = function() {
145 if (this.element_) {
146 var firstHeader = this.element_.querySelector('th');
147 var rows = this.element_.querySelector('tbody').querySelectorAll('tr');
148
149 if (this.element_.classList.contains(this.CssClasses_.SELECTABLE)) {
150 var th = document.createElement('th');
151 var headerCheckbox = this.createCheckbox_(null, rows);
152 th.appendChild(headerCheckbox);
153 firstHeader.parentElement.insertBefore(th, firstHeader);
154
155 for (var i = 0; i < rows.length; i++) {
156 var firstCell = rows[i].querySelector('td');
157 if (firstCell) {
158 var td = document.createElement('td');
159 var rowCheckbox = this.createCheckbox_(rows[i]);
160 td.appendChild(rowCheckbox);
161 rows[i].insertBefore(td, firstCell);
162 }
163 }
164 }
165
166 this.element_.classList.add(this.CssClasses_.IS_UPGRADED);
167 }
168 };
169
170 // The component registers itself. It can assume componentHandler is available
171 // in the global scope.
172 componentHandler.register({
173 constructor: MaterialDataTable,
174 classAsString: 'MaterialDataTable',
175 cssClass: goog.getCssName( 'mdl-js-data-table' )
176 });
