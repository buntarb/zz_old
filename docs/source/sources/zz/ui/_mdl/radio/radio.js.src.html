28(function() {
29 'use strict';

31 /**
32 * Class constructor for Radio MDL component.
33 * Implements MDL component design pattern defined at:
34 * https://github.com/jasonmayes/mdl-component-design-pattern
35 *
36 * @constructor
37 * @param {HTMLElement} element The element that will be upgraded.

39 var MaterialRadio = function MaterialRadio(element) {
40 this.element_ = element;
41
42 // Initialize instance.
43 this.init();
44 };
45 window['MaterialRadio'] = MaterialRadio;

47 /**
48 * Store constants in one place so they can be updated easily.
49 *
50 * @enum {string | number}
51 * @private
52 */
53 MaterialRadio.prototype.Constant_ = {
54 TINY_TIMEOUT: 0.001
55 };
56
57 /**
58 * Store strings for class names defined by this component that are used in
59 * JavaScript. This allows us to simply change it in one place should we
60 * decide to modify at a later date.
61 *
62 * @enum {string}
63 * @private
64 */
65 MaterialRadio.prototype.CssClasses_ = {
66 IS_FOCUSED: goog.getCssName( 'is-focused' ),
67 IS_DISABLED: goog.getCssName( 'is-disabled' ),
68 IS_CHECKED: goog.getCssName( 'is-checked' ),
69 IS_UPGRADED: goog.getCssName( 'is-upgraded' ),
70 JS_RADIO: goog.getCssName( 'mdl-js-radio' ),
71 RADIO_BTN: goog.getCssName( 'mdl-radio__button' ),
72 RADIO_OUTER_CIRCLE: goog.getCssName( 'mdl-radio__outer-circle' ),
73 RADIO_INNER_CIRCLE: goog.getCssName( 'mdl-radio__inner-circle' ),
74 RIPPLE_EFFECT: goog.getCssName( 'mdl-js-ripple-effect' ),
75 RIPPLE_IGNORE_EVENTS: goog.getCssName( 'mdl-js-ripple-effect--ignore-events' ),
76 RIPPLE_CONTAINER: goog.getCssName( 'mdl-radio__ripple-container' ),
77 RIPPLE_CENTER: goog.getCssName( 'mdl-ripple--center' ),
78 RIPPLE: goog.getCssName( 'mdl-ripple' )
79 };

81 /**
82 * Handle change of state.
83 *
84 * @param {Event} event The event that fired.
85 * @private
86 */
87 MaterialRadio.prototype.onChange_ = function(event) {
88 // Since other radio buttons don't get change events, we need to look for
89 // them to update their classes.
90 var radios = document.getElementsByClassName(this.CssClasses_.JS_RADIO);
91 for (var i = 0; i < radios.length; i++) {
92 var button = radios[i].querySelector('.' + this.CssClasses_.RADIO_BTN);
93 // Different name == different group, so no point updating those.
94 if (button.getAttribute('name') === this.btnElement_.getAttribute('name')) {
95 radios[i]['MaterialRadio'].updateClasses_();
96 }
97 }
98 };
99
100 /**
101 * Handle focus.
102 *
103 * @param {Event} event The event that fired.
104 * @private
105 */
106 MaterialRadio.prototype.onFocus_ = function(event) {
107 this.element_.classList.add(this.CssClasses_.IS_FOCUSED);
108 };
109
110 /**
111 * Handle lost focus.
112 *
113 * @param {Event} event The event that fired.
114 * @private
115 */
116 MaterialRadio.prototype.onBlur_ = function(event) {
117 this.element_.classList.remove(this.CssClasses_.IS_FOCUSED);
118 };
119
120 /**
121 * Handle mouseup.
122 *
123 * @param {Event} event The event that fired.
124 * @private
125 */
126 MaterialRadio.prototype.onMouseup_ = function(event) {
127 this.blur_();
128 };
129
130 /**
131 * Update classes.
132 *
133 * @private
134 */
135 MaterialRadio.prototype.updateClasses_ = function() {
136 this.checkDisabled();
137 this.checkToggleState();
138 };
139
140 /**
141 * Add blur.
142 *
143 * @private
144 */
145 MaterialRadio.prototype.blur_ = function() {
146
147 // TODO: figure out why there's a focus event being fired after our blur,
148 // so that we can avoid this hack.
149 window.setTimeout(function() {
150 this.btnElement_.blur();
151 }.bind(this), /** @type {number} */ (this.Constant_.TINY_TIMEOUT));
152 };

156 /**
157 * Check the components disabled state.
158 *
159 * @public
160 */
161 MaterialRadio.prototype.checkDisabled = function() {
162 if (this.btnElement_.disabled) {
163 this.element_.classList.add(this.CssClasses_.IS_DISABLED);
164 } else {
165 this.element_.classList.remove(this.CssClasses_.IS_DISABLED);
166 }
167 };
168 MaterialRadio.prototype['checkDisabled'] = MaterialRadio.prototype.checkDisabled;
169
170 /**
171 * Check the components toggled state.
172 *
173 * @public
174 */
175 MaterialRadio.prototype.checkToggleState = function() {
176 if (this.btnElement_.checked) {
177 this.element_.classList.add(this.CssClasses_.IS_CHECKED);
178 } else {
179 this.element_.classList.remove(this.CssClasses_.IS_CHECKED);
180 }
181 };
182 MaterialRadio.prototype['checkToggleState'] = MaterialRadio.prototype.checkToggleState;
183

228 /**
229 * Initialize element.
230 */
231 MaterialRadio.prototype.init = function() {
232 if (this.element_) {
233 this.btnElement_ = this.element_.querySelector('.' + this.CssClasses_.RADIO_BTN);
234
235 this.boundChangeHandler_ = this.onChange_.bind(this);
236 this.boundFocusHandler_ = this.onChange_.bind(this);
237 this.boundBlurHandler_ = this.onBlur_.bind(this);
238 this.boundMouseUpHandler_ = this.onMouseup_.bind(this);
239
240 var outerCircle = document.createElement('span');
241 outerCircle.classList.add(this.CssClasses_.RADIO_OUTER_CIRCLE);
242
243 var innerCircle = document.createElement('span');
244 innerCircle.classList.add(this.CssClasses_.RADIO_INNER_CIRCLE);
245
246 this.element_.appendChild(outerCircle);
247 this.element_.appendChild(innerCircle);
248
249 var rippleContainer;
250 if (this.element_.classList.contains( this.CssClasses_.RIPPLE_EFFECT )) {
251 this.element_.classList.add( this.CssClasses_.RIPPLE_IGNORE_EVENTS );
252 rippleContainer = document.createElement('span');
253 rippleContainer.classList.add( this.CssClasses_.RIPPLE_CONTAINER );
254 rippleContainer.classList.add( this.CssClasses_.RIPPLE_EFFECT );
255 rippleContainer.classList.add( this.CssClasses_.RIPPLE_CENTER );
256 rippleContainer.addEventListener('mouseup', this.boundMouseUpHandler_);
257
258 var ripple = document.createElement('span');
259 ripple.classList.add(this.CssClasses_.RIPPLE);
260
261 rippleContainer.appendChild(ripple);
262 this.element_.appendChild(rippleContainer);
263 }
264
265 this.btnElement_.addEventListener('change', this.boundChangeHandler_);
266 this.btnElement_.addEventListener('focus', this.boundFocusHandler_);
267 this.btnElement_.addEventListener('blur', this.boundBlurHandler_);
268 this.element_.addEventListener('mouseup', this.boundMouseUpHandler_);
269
270 this.updateClasses_();
271 this.element_.classList.add(this.CssClasses_.IS_UPGRADED);
272 }
273 };
274
275 /**
276 * Downgrade the element.
277 *
278 * @private
279 */
280 MaterialRadio.prototype.mdlDowngrade_ = function() {
281 var rippleContainer = this.element_.querySelector('.' + this.CssClasses_.RIPPLE_CONTAINER);
282 this.btnElement_.removeEventListener('change', this.boundChangeHandler_);
283 this.btnElement_.removeEventListener('focus', this.boundFocusHandler_);
284 this.btnElement_.removeEventListener('blur', this.boundBlurHandler_);
285 this.element_.removeEventListener('mouseup', this.boundMouseUpHandler_);
286 if (rippleContainer) {
287 rippleContainer.removeEventListener('mouseup', this.boundMouseUpHandler_);
288 this.element_.removeChild(rippleContainer);
289 }
290 };

292 /**

297 MaterialRadio.prototype.mdlDowngrade = MaterialRadio.prototype.mdlDowngrade_;
298 MaterialRadio.prototype['mdlDowngrade'] = MaterialRadio.prototype.mdlDowngrade;
