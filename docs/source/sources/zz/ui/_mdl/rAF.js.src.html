14goog.provide( 'zz.ui.mdl.requestAnimationFrame' );
15goog.provide( 'zz.ui.mdl.cancelAnimationFrame' );
16
17(function() {
18'use strict';
19
20if (!Date.now) {
21 /**
22 * Date.now polyfill.
23 * @return {number} the current Date
24 */
25 Date.now = function() { return new Date().getTime(); };
26 Date['now'] = Date.now;
27}
28
29var vendors = ['webkit', 'moz'];
30for (var i = 0; i < vendors.length && !window.requestAnimationFrame; ++i) {
31 var vp = vendors[i];
32 window.requestAnimationFrame = window[vp + 'RequestAnimationFrame'];
33 window.cancelAnimationFrame = (window[vp + 'CancelAnimationFrame'] ||
34 window[vp + 'CancelRequestAnimationFrame']);
35 window['requestAnimationFrame'] = window.requestAnimationFrame;
36 window['cancelAnimationFrame'] = window.cancelAnimationFrame;
37}
38
39if (/iP(ad|hone|od).*OS 6/.test(window.navigator.userAgent) || !window.requestAnimationFrame || !window.cancelAnimationFrame) {
40 var lastTime = 0;
41 /**
42 * requestAnimationFrame polyfill.
43 * @param {!Function} callback the callback function.
44 */
45 window.requestAnimationFrame = function(callback) {
46 var now = Date.now();
47 var nextTime = Math.max(lastTime + 16, now);
48 return setTimeout(function() { callback(lastTime = nextTime); },
49 nextTime - now);
50 };
51 window.cancelAnimationFrame = clearTimeout;
52 window['requestAnimationFrame'] = window.requestAnimationFrame;
53 window['cancelAnimationFrame'] = window.cancelAnimationFrame;
54}
55
56 /********************************************************************************************************************
57 * Definition section *
58 ********************************************************************************************************************/
59 zz.ui.mdl.requestAnimationFrame = requestAnimationFrame;
60 zz.ui.mdl.cancelAnimationFrame = cancelAnimationFrame;
61
62})();
