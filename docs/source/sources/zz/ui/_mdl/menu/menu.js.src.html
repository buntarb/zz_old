198 /**
199 * Handles a click on the "for" element, by positioning the menu and then
200 * toggling it.
201 *
202 * @param {Event} evt The event that fired.
203 * @private
204 */
205 MaterialMenu.prototype.handleForClick_ = function(evt) {
206 if (this.element_ && this.forElement_) {
207 var rect = this.forElement_.getBoundingClientRect();
208 var forRect = this.forElement_.parentElement.getBoundingClientRect();
209
210 if (this.element_.classList.contains(this.CssClasses_.UNALIGNED)) {
211 // Do not position the menu automatically. Requires the developer to
212 // manually specify position.
213 } else if (this.element_.classList.contains(
214 this.CssClasses_.BOTTOM_RIGHT)) {
215 // Position below the "for" element, aligned to its right.
216 this.container_.style.right = (forRect.right - rect.right) + 'px';
217 this.container_.style.top =
218 this.forElement_.offsetTop + this.forElement_.offsetHeight + 'px';
219 } else if (this.element_.classList.contains(this.CssClasses_.TOP_LEFT)) {
220 // Position above the "for" element, aligned to its left.
221 this.container_.style.left = this.forElement_.offsetLeft + 'px';
222 this.container_.style.bottom = (forRect.bottom - rect.top) + 'px';
223 } else if (this.element_.classList.contains(this.CssClasses_.TOP_RIGHT)) {
224 // Position above the "for" element, aligned to its right.
225 this.container_.style.right = (forRect.right - rect.right) + 'px';
226 this.container_.style.bottom = (forRect.bottom - rect.top) + 'px';
227 } else {
228 // Default: position below the "for" element, aligned to its left.
229 this.container_.style.left = this.forElement_.offsetLeft + 'px';
230 this.container_.style.top =
231 this.forElement_.offsetTop + this.forElement_.offsetHeight + 'px';
232 }
233 }
234
235 this.toggle(evt);
236 };

238 /**
239 * Handles a keyboard event on the "for" element.
240 *
241 * @param {Event} evt The event that fired.
242 * @private
243 */
244 MaterialMenu.prototype.handleForKeyboardEvent_ = function(evt) {
245 if (this.element_ && this.container_ && this.forElement_) {
246 var items = this.element_.querySelectorAll('.' + this.CssClasses_.ITEM +
247 ':not([disabled])');
248
249 if (items && items.length > 0 &&
250 this.container_.classList.contains(this.CssClasses_.IS_VISIBLE)) {
251 if (evt.keyCode === this.Keycodes_.UP_ARROW) {
252 evt.preventDefault();
253 items[items.length - 1].focus();
254 } else if (evt.keyCode === this.Keycodes_.DOWN_ARROW) {
255 evt.preventDefault();
256 items[0].focus();
257 }
258 }
259 }
260 };

262 /**
263 * Handles a keyboard event on an item.
264 *
265 * @param {Event} evt The event that fired.
266 * @private
267 */
268 MaterialMenu.prototype.handleItemKeyboardEvent_ = function(evt) {
269 if (this.element_ && this.container_) {
270 var items = this.element_.querySelectorAll('.' + this.CssClasses_.ITEM +
271 ':not([disabled])');
272
273 if (items && items.length > 0 &&
274 this.container_.classList.contains(this.CssClasses_.IS_VISIBLE)) {
275 var currentIndex = Array.prototype.slice.call(items).indexOf(evt.target);
276
277 if (evt.keyCode === this.Keycodes_.UP_ARROW) {
278 evt.preventDefault();
279 if (currentIndex > 0) {
280 items[currentIndex - 1].focus();
281 } else {
282 items[items.length - 1].focus();
283 }
284 } else if (evt.keyCode === this.Keycodes_.DOWN_ARROW) {
285 evt.preventDefault();
286 if (items.length > currentIndex + 1) {
287 items[currentIndex + 1].focus();
288 } else {
289 items[0].focus();
290 }
291 } else if (evt.keyCode === this.Keycodes_.SPACE ||
292 evt.keyCode === this.Keycodes_.ENTER) {
293 evt.preventDefault();
294 // Send mousedown and mouseup to trigger ripple.
295 var e = new MouseEvent('mousedown');
296 evt.target.dispatchEvent(e);
297 e = new MouseEvent('mouseup');
298 evt.target.dispatchEvent(e);
299 // Send click.
300 evt.target.click();
301 } else if (evt.keyCode === this.Keycodes_.ESCAPE) {
302 evt.preventDefault();
303 this.hide();
304 }
305 }
306 }
307 };

309 /**
310 * Handles a click event on an item.
311 *
312 * @param {Event} evt The event that fired.
313 * @private
314 */
315 MaterialMenu.prototype.handleItemClick_ = function(evt) {
316 if (evt.target.hasAttribute('disabled')) {
317 evt.stopPropagation();
318 } else {
319 // Wait some time before closing menu, so the user can see the ripple.
320 this.closing_ = true;
321 window.setTimeout(function(evt) {
322 this.hide();
323 this.closing_ = false;
324 }.bind(this), /** @type {number} */ (this.Constant_.CLOSE_TIMEOUT));
325 }
326 };

328 /**
329 * Calculates the initial clip (for opening the menu) or final clip (for closing
330 * it), and applies it. This allows us to animate from or to the correct point,
331 * that is, the point it's aligned to in the "for" element.
332 *
333 * @param {number} height Height of the clip rectangle
334 * @param {number} width Width of the clip rectangle
335 * @private
336 */
337 MaterialMenu.prototype.applyClip_ = function(height, width) {
338 if (this.element_.classList.contains(this.CssClasses_.UNALIGNED)) {
339 // Do not clip.
340 this.element_.style.clip = '';
341 } else if (this.element_.classList.contains(this.CssClasses_.BOTTOM_RIGHT)) {
342 // Clip to the top right corner of the menu.
343 this.element_.style.clip =
344 'rect(0 ' + width + 'px ' + '0 ' + width + 'px)';
345 } else if (this.element_.classList.contains(this.CssClasses_.TOP_LEFT)) {
346 // Clip to the bottom left corner of the menu.
347 this.element_.style.clip =
348 'rect(' + height + 'px 0 ' + height + 'px 0)';
349 } else if (this.element_.classList.contains(this.CssClasses_.TOP_RIGHT)) {
350 // Clip to the bottom right corner of the menu.
351 this.element_.style.clip = 'rect(' + height + 'px ' + width + 'px ' +
352 height + 'px ' + width + 'px)';
353 } else {
354 // Default: do not clip (same as clipping to the top left corner).
355 this.element_.style.clip = '';
356 }
357 };

359 /**

364 MaterialMenu.prototype.addAnimationEndListener_ = function() {
365 var cleanup = function() {
366 this.element_.removeEventListener('transitionend', cleanup);
367 this.element_.removeEventListener('webkitTransitionEnd', cleanup);
368 this.element_.classList.remove(this.CssClasses_.IS_ANIMATING);

373 this.element_.addEventListener('webkitTransitionEnd', cleanup);

376 /**
377 * Displays the menu.
378 *
379 * @public
380 */
381 MaterialMenu.prototype.show = function(evt) {
382 if (this.element_ && this.container_ && this.outline_) {
383 // Measure the inner element.
384 var height = this.element_.getBoundingClientRect().height;
385 var width = this.element_.getBoundingClientRect().width;
386
387 // Apply the inner element's size to the container and outline.
388 this.container_.style.width = width + 'px';
389 this.container_.style.height = height + 'px';
390 this.outline_.style.width = width + 'px';
391 this.outline_.style.height = height + 'px';
392
393 var transitionDuration = this.Constant_.TRANSITION_DURATION_SECONDS *
394 this.Constant_.TRANSITION_DURATION_FRACTION;
395
396 // Calculate transition delays for individual menu items, so that they fade
397 // in one at a time.
398 var items = this.element_.querySelectorAll('.' + this.CssClasses_.ITEM);
399 for (var i = 0; i < items.length; i++) {
400 var itemDelay = null;
401 if (this.element_.classList.contains(this.CssClasses_.TOP_LEFT) ||
402 this.element_.classList.contains(this.CssClasses_.TOP_RIGHT)) {
403 itemDelay = ((height - items[i].offsetTop - items[i].offsetHeight) /
404 height * transitionDuration) + 's';
405 } else {
406 itemDelay = (items[i].offsetTop / height * transitionDuration) + 's';
407 }
408 items[i].style.transitionDelay = itemDelay;
409 }
410
411 // Apply the initial clip to the text before we start animating.
412 this.applyClip_(height, width);
413
414 // Wait for the next frame, turn on animation, and apply the final clip.
415 // Also make it visible. This triggers the transitions.
416 window.requestAnimationFrame(function() {
417 this.element_.classList.add(this.CssClasses_.IS_ANIMATING);
418 this.element_.style.clip = 'rect(0 ' + width + 'px ' + height + 'px 0)';
419 this.container_.classList.add(this.CssClasses_.IS_VISIBLE);
420 }.bind(this));
421
422 // Clean up after the animation is complete.
423 this.addAnimationEndListener_();
424
425 // Add a click listener to the document, to close the menu.
426 var callback = function(e) {
427 // Check to see if the document is processing the same event that
428 // displayed the menu in the first place. If so, do nothing.
429 // Also check to see if the menu is in the process of closing itself, and
430 // do nothing in that case.
431 // Also check if the clicked element is a menu item
432 // if so, do nothing.
433 if (e !== evt && !this.closing_ && e.target.parentNode !== this.element_) {
434 document.removeEventListener('click', callback);
435 this.hide();
436 }
437 }.bind(this);
438 document.addEventListener('click', callback);
439 }
440 };
441 MaterialMenu.prototype['show'] = MaterialMenu.prototype.show;

443 /**
444 * Hides the menu.
445 *
446 * @public
447 */
448 MaterialMenu.prototype.hide = function() {
449 if (this.element_ && this.container_ && this.outline_) {
450 var items = this.element_.querySelectorAll('.' + this.CssClasses_.ITEM);
451
452 // Remove all transition delays; menu items fade out concurrently.
453 for (var i = 0; i < items.length; i++) {
454 items[i].style.transitionDelay = null;
455 }
456
457 // Measure the inner element.
458 var rect = this.element_.getBoundingClientRect();
459 var height = rect.height;
460 var width = rect.width;
461
462 // Turn on animation, and apply the final clip. Also make invisible.
463 // This triggers the transitions.
464 this.element_.classList.add(this.CssClasses_.IS_ANIMATING);
465 this.applyClip_(height, width);
466 this.container_.classList.remove(this.CssClasses_.IS_VISIBLE);
467
468 // Clean up after the animation is complete.
469 this.addAnimationEndListener_();
470 }
471 };
