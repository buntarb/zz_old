199/**
200 * Calling protected zz.mvc.view.BaseView#modelChangedInternal to process model changes and update current view.
201 * @param {zz.mvc.model.Message} message
202 * @final
203 */
204zz.mvc.view.BaseView.prototype.modelChanged = function( message ){
205
206 var model = this.getModel( );
207 if( goog.isDefAndNotNull( model ) ){
208
209 if( goog.isDefAndNotNull( model.datafield ) ){
210
211 if( model.datarow.getUid( ) === message.datarow.getUid( ) ){
212
213 this.modelChangedInternal( message );
214 }
215 }else{
216
217 this.modelChangedInternal( message );
218 }
219 }
220};

222/**
223 * Update view to process model changes. This method need to be override by child.
224 * @param {zz.mvc.model.Message} message
225 * @protected
226 */
227zz.mvc.view.BaseView.prototype.modelChangedInternal = function( message ){
228
229 var jsonFormatter = new goog.format.JsonPrettyPrinter( new goog.format.JsonPrettyPrinter.TextDelimiters( ) );
230 var content = jsonFormatter.format( goog.json.serialize( message.dataset.serializeDataset( true ) ) );
231 var element = goog.dom.createElement( 'pre' );
232 if( this.getContentElement( ) ){
233
234 goog.dom.setTextContent( element, content );
235 goog.dom.removeChildren( this.getElement( ) );
236 goog.dom.insertChildAt( this.getElement( ), element, 0 );
237 }
238};
